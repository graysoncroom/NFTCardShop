About the Collection

NFT Card shop is a new and unique concept ready to disrupt the collectable trading card world forever. Our platform and backend contract ties your ownership of the virtual NFT card to a physical graded card. Having your NFT linked to physical asset opens doors to numerous benefits. Your NFT’S value will be backed by a physical asset which gives security in this highly volatile market. Within our secure vault we store all of the physical cards which can be sent upon request. By leaving your physical card in our vault your investment remains insured and you no longer have the liability of storing the physical asset. Our vault system also allows for quick transfer of ownership and provides confidence in the authenticity of your purchase.

This is the vault

If you have any Cards, they will appear here. You may request to hold your physical card if you wish, as the holder of the NFT is the rightful owner of the physical card as well.

The Contract

This is our contract. Built on top of ERC721, the industry standard for non-fungible tokens. The assets are stored on IPFS, so that your NFT will remain with you no matter what happens to us.

Our Contract on Etherscan 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.4;

import "@openzeppelin/contracts@4.5.0/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts@4.5.0/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts@4.5.0/access/Ownable.sol";
import "@openzeppelin/contracts@4.5.0/utils/Counters.sol";

contract NFTCardShop is ERC721, ERC721URIStorage, Ownable {
    using Counters for Counters.Counter;

    Counters.Counter private _tokenIdCounter;

    mapping(uint256 => bool) tokenIsInVault;

    constructor() ERC721("NFTCardShop", "CaSh") {}

    function _baseURI() internal pure override returns (string memory) {
        return "https://gateway.pinata.cloud/ipfs/";
    }

    function safeMint(address to, string memory uri) public onlyOwner {
        uint256 tokenId = _tokenIdCounter.current();
        _tokenIdCounter.increment();
        _safeMint(to, tokenId);
        _setTokenURI(tokenId, uri);
        tokenIsInVault[tokenId] = true;
    }

    function _beforeTokenTransfer(
        address from, 
        address to, 
        uint256 tokenId
    ) internal override(ERC721) {
        require(tokenIsInVault[tokenId]);
    }

    // will be used to update the NFT with a "Card Secured In Vault" checkmark
    function setTokenURI(uint256 tokenId, string memory _tokenURI) public onlyOwner {
        super._setTokenURI(tokenId, _tokenURI);
    }

    // The following functions are overrides required by Solidity.

    function _burn(uint256 tokenId) internal override(ERC721, ERC721URIStorage) {
        super._burn(tokenId);
    }

    function tokenURI(uint256 tokenId) public
        view
        override(ERC721, ERC721URIStorage)
        returns (string memory)
    {
        return super.tokenURI(tokenId);
    }
}
